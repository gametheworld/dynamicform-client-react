Switch

            
                
                {
                    "type" : "switch",
                    "name" : "switch",
                    "path" : "ProfessionalInformation.Switch",
                    "label" : "开关",
                    "isicon" : true,
                    "defaultvalue" : true,
                    "rules" : [
                        {
                            "required" : true,
                            "message" : "请选择"
                        }
                    ]，
                    "hidden":false
                    "disabled":false
                }

type: 元素类型

name: instance name - 唯一

path: 数据提交json格式

label: 输入框label

disabled:是否禁用组件

defaultvalue:默认值

rules: Array ,验证规则

isicon:是否开启Icon图标，目前该值无效

disabled:是否禁用组件

hidden: 是否显示，不显示：true，显示：false；默认false
