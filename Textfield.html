Textfield

            
                
                    {
                        "type" : "textfield",
                        "name" : "fullname",
                        "path" : "ProfessionalInformation.fullname",
                        "inputType" : "text",
                        "label" : "姓名",
                        "defaultvalue" : "张三",
                        "placeholder" : "请输入姓名",
                        "addonBefore" : "",
                        "addonAfter" : "",
                        "rules" : [
                            {
                                "required" : true,
                                "message" : "姓名必填"
                            },
                            {
                                "pattern" : "/^\\d+$/",
                                "message" : "请输入数字"
                            },
                            {
                                 "type": "email",
                                 "message": "The input is not valid E-mail!"
                            }
                        ]，
                        "hidden":false
                        "disabled":false
                    }

type: 元素类型

path: 数据提交json格式

name: instance name - 唯一

inputType: 输入框类型- 参考html input type

label: 输入框label

defaultvalue:默认值

placeholder: 输入框placeholder

addonBefore: 输入框前置字符

addonAfter: 输入框后置字符

rules: Array ,验证规则

disabled:是否禁用组件

hidden: 是否显示，不显示：true，显示：false；默认false
