TimePicker

            
                
                {
                    "type" : "timepicker",
                    "name" : "timepicker",
                    "path" : "ProfessionalInformation.ConpanyInDate",
                    "label" : "时间",
                    "placeholder" : "时间",
                    "format" : "HH:mm:ss"，
                    "hidden":false
                    "disabled":false
                    "defaultvalue" : "23:24:12",
                    "use12hours" : false,
                    "rules" : [
                        {
                            "required" : true,
                            "message" : "请输入时间"
                        }
                    ]
                }

type: 元素类型

name: instance name - 唯一

path: 数据提交json格式

label: 输入框label

format: 日期格式

disabled:是否禁用组件

hidden: 是否显示，不显示：true，显示：false；默认false

defaultvalue:默认值

placeholder: 输入框placeholder

rules: Array ,验证规则

use12hours:使用 12 小时制，为 true 时 format 默认为 h:mm:ss a

更多展示的日期格式，配置参考 moment
