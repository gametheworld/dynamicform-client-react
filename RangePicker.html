RangePicker

            
                
                {
                    "type" : "rangepicker",
                    "name" : "rangepicker",
                    "path" : "ProfessionalInformation.Rangepicker",
                    "label" : "选择日期范围",
                    "placeholder" : "选择日期范围",
                    "format" : "YYYY-MM-DD",
                    "defaultvalue" : {
                        "starttime" : "2017-04-01",
                        "endtime" : "2017-05-17"
                    },
                    "showtime" : false,
                    "hidden": false,
                    "disabled":false,
                    "rules" : [
                        {
                            "required" : true,
                            "message" : "选择日期范围""
                        }
                    ]
                }

type: 元素类型

name: instance name - 唯一

path: 数据提交json格式

label: 输入框label

format: 日期格式

defaultvalue:

::starttime:起始日期默认值

::endtime:结束日期默认值

showtime:增加时间选择功能,默认

disabled:是否禁用组件

hidden: 是否显示，不显示：true，显示：false；默认false

placeholder: 输入框placeholder

rules: Array ,验证规则

更多展示的日期格式，配置参考 moment
