DynamicControl

            
                
                {
                    "type" : "DynamicControl",
                    "label" : "联系人",
                    "name" : "containerInfo",
                    "title" : [
                        "家庭联系人1",
                        "家庭联系人2",
                        "紧急联系人",
                        "工作证明人"
                    ],
                    "defaultCount" : 4,
                    "template" : {
                        "type" : "container",
                        "name" : "ContactDetail",
                        "label" : "家庭联系人1",
                        "components" : [ ]
                    }
                }

type: 元素类型

name: instance name - 唯一

label: 输入框label

title: 模版title名称

defaultCount: 模版默认显示个数

template: 模版元素
