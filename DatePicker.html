DatePicker

            
                
                {
                    "type" : "datepicker",
                    "name" : "datepicker",
                    "path" : "ProfessionalInformation.ConpanyInDate",
                    "label" : "日期",
                    "format" : "YYYY-MM-DD"，
                    "hidden":false
                    "disabled":false
                    "defaultvalue" : "2017-05-17",
                    "showtime" : false,
                    "placeholder" : "请输入日期",
                    "rules" : [
                        {
                            "required" : true,
                            "message" : "日期不能为空"
                        }
                    ],
                    "style" : {
                        "width" : "300px"
                    },
                    "popupstyle" : {
                        "width" : "500px"
                    }
                }

type: 元素类型

name: instance name - 唯一

path: 数据提交json格式

label: 输入框label

format: 日期格式

disabled:是否禁用组件

hidden: 是否显示，不显示：true，显示：false；默认false

defaultvalue:默认值

showtime:是否显示时间(如果为True，format:YYYY-MM-DD kk:mm:ss)

placeholder: 输入框placeholder

rules: Array ,验证规则

style:自定义样式

popupstyle:自定义的弹出窗口的样式

更多展示的日期格式，配置参考 moment
